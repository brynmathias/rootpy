[docs]def scan(subnet='192.168.1', port=50000):

    s = socket.socket()
    s.settimeout(0.1)
    for ip in xrange(1, 256):
        addr = "%s.%s" % (subnet, ip)
        try:
            s.connect((addr, port))
            yield addr
        except:
            pass
    s.close()
